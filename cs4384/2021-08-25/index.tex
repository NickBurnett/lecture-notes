  \section{Deterministic Finite Automata}
  \begin{itemize}
    \item Finite automata, single-directional
    \item Q = set of states (set)
    \item \begin{math}
      \Sigma = alphabet (allowed\_input)
    \end{math}
    \item \begin{math}
      \delta = transitions (Q*\Sigma -> Q)
    \end{math}
    \item F(0) = start state
    \item F = set of accepted states
    \item Arrow going to a state from nothing is defined as the start state
  \end{itemize}
